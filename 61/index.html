Rotate List

Given the head of a linked list, rotate the list to the right by k places.

Example 1:

Input: head = [1,2,3,4,5], k = 2 Output: [4,5,1,2,3]

Example 2:

Input: head = [0,1,2], k = 4 Output: [2,0,1]

Constraints:

  • The number of nodes in the list is in the range [0, 500].
  • -100 <= Node.val <= 100
  • 0 <= k <= 2 * 109

pub struct Solution;

impl Solution {
    pub fn rotate_right(head: Option>, k: i32) -> Option> {
        let mut len = 0;
        let mut current = head;
        while let Some(node) = current {
            len += 1;
            current = node.next;
        }

        let k = k % len;
        if k == 0 {
            return head;
        }

        let mut current = head;
        let mut prev = None;
        for _ in 0..len - k {
            prev = current;
            current = current.as_ref().unwrap().next;
        }

        if let Some(prev) = prev {
            prev.next = head;
        }

        current.as_mut().unwrap().next = None;
        head
    }
}

pub struct ListNode {
    pub val: i32,
    pub next: Option>,
}

impl ListNode {
    #[inline]
    fn new(val: i32) -> Self {
        ListNode {
            next: None,
            val,
        }
    }
}

impl Solution {
    pub fn rotate_right(head: Option>, k: i32) -> Option> {
        if head.is_none() {
            return head;
        }

        let mut length = 0;
        let mut current = head;
        while current.is_some() {
            length += 1;
            current = current.unwrap().next;
        }

        let k = k % length;
        if k == 0 {
            return head;
        }

        let mut new_head = head;
        for _ in 0..(length - k) {
            new_head = new_head.unwrap().next;
        }

        let mut current = head;
        while current.is_some() {
            let next = current.unwrap().next;
            current.unwrap().next = new_head;
            current = next;
        }

        new_head
    }
}
